# Include directories

    PRIVATE

    PUBLIC
    ${DEFAULT_INCLUDE_DIRECTORIES}

)

    PUBLIC
    ${DEFAULT_LIBRARIES}

    Qt5::Sql
    Qt5::Widgets
    Qt5::WebSockets

)
